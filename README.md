```bash
npm install --save react-carousel-prototipe
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-carousel-prototipe'
import 'react-carousel-prototipe/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
